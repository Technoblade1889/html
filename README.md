<!-- 以下是一些典型的双标签 -->
<p>这是一个段落</p>
<h1>这是一个标题</h1>
<h2>这是第二个标题</h2>
<a href="#">这是一个超链接</a>

<!-- 以下是一些典型的单标签 -->
 <input type="text">
 <br>
 <hr>

<!-- 他们的区别：双标签要存内容，单标签只起指令作用比如换行 -->

<!-- 下面是html的框架，所有的网页布局都会在里面完成 -->
<!-- 只要输入！就可以了，不然要打完很麻烦 -->
<!DOCTYPE html>
<html lang="en">
<!-- 这里是网页的头部，用于存放标题 -->
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>HTML练习1-标签</title>
</head>
<!-- 这里是网页的内容部分 -->
<body>
<!--我将在以下部分简单对我目前所学到的标签进行罗列  -->
<h1>标题1</h1><!-- 标题格式可以从1-6 -->
<p>段落</p>
<p>更改文本：<b>字体加粗</b>、<i>斜体</i>、<s>删除线</s>、<u>下划线</u></p><!-- b也可以换成strong -->

<!-- 列表 -->
<ul>
    <li>无序</li>
</ul>

<ol>
    <li>有序</li>
</ol>


</body>

</html>
